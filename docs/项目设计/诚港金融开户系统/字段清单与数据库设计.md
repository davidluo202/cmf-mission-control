## 🗄️ 数据库Schema设计

### 主表：corporate_accounts (机构账户主表)

```sql
CREATE TABLE corporate_accounts (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  account_number VARCHAR(50) UNIQUE NOT NULL COMMENT '账户号码',
  
  -- Section A: Client Information
  company_name_en VARCHAR(255) NOT NULL,
  company_name_zh VARCHAR(255),
  nature_of_entity VARCHAR(100) NOT NULL,
  nature_of_business VARCHAR(100) NOT NULL,
  country_of_incorporation VARCHAR(100) NOT NULL,
  date_of_incorporation DATE NOT NULL,
  certificate_of_incorporation_no VARCHAR(100) NOT NULL,
  business_registration_no VARCHAR(100),
  jurisdiction_of_residence VARCHAR(100),
  
  registered_address TEXT NOT NULL,
  business_address TEXT,
  office_phone VARCHAR(20) NOT NULL,
  fax_no VARCHAR(20),
  
  contact_person_name VARCHAR(100) NOT NULL,
  contact_person_title VARCHAR(50) NOT NULL,
  contact_person_phone VARCHAR(20) NOT NULL,
  contact_person_email VARCHAR(100) NOT NULL,
  
  -- Section C: Financial Profile
  authorized_share_capital DECIMAL(15,2) NOT NULL,
  issued_paid_up_capital DECIMAL(15,2) NOT NULL,
  initial_source_of_wealth JSON NOT NULL COMMENT '多选字段JSON存储',
  net_asset_value VARCHAR(20) NOT NULL,
  net_asset_audit_year VARCHAR(20) NOT NULL,
  profit_after_tax VARCHAR(20) NOT NULL,
  profit_audit_year VARCHAR(20) NOT NULL,
  
  -- Section D: Investment Profile
  investment_objective JSON NOT NULL,
  estimated_investment_amount VARCHAR(20) NOT NULL,
  risk_volatility VARCHAR(20) NOT NULL,
  investment_experience VARCHAR(20) NOT NULL,
  derivatives_knowledge JSON NOT NULL,
  experienced_products JSON,
  asset_items JSON,
  
  -- 开户流程状态
  status VARCHAR(20) NOT NULL DEFAULT 'draft' COMMENT 'draft/submitted/approved/rejected',
  current_case INT DEFAULT 1 COMMENT '当前完成的步骤',
  
  -- 风险评估结果 (来自 CMF013)
  risk_profile VARCHAR(20) COMMENT 'Conservative/Moderate/Aggressive',
  risk_questionnaire_completed_at TIMESTAMP,
  
  -- 审计字段
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  submitted_at TIMESTAMP,
  approved_at TIMESTAMP,
  approved_by VARCHAR(100),
  
  INDEX idx_status (status),
  INDEX idx_company_name (company_name_en),
  INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

### 子表1：corporate_settlement_accounts (结算银行账户)

```sql
CREATE TABLE corporate_settlement_accounts (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  corporate_account_id BIGINT NOT NULL,
  
  bank_name VARCHAR(100) NOT NULL,
  account_number VARCHAR(50) NOT NULL,
  currency VARCHAR(10) NOT NULL,
  swift_code VARCHAR(20),
  is_primary BOOLEAN DEFAULT FALSE COMMENT '是否为主账户',
  
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (corporate_account_id) REFERENCES corporate_accounts(id) ON DELETE CASCADE,
  INDEX idx_corporate_account (corporate_account_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 子表2：corporate_connected_parties (关联人士)

```sql
CREATE TABLE corporate_connected_parties (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  corporate_account_id BIGINT NOT NULL,
  
  party_type VARCHAR(20) NOT NULL COMMENT 'director/beneficial_owner/authorized_signatory',
  
  -- 个人信息
  full_name_en VARCHAR(100) NOT NULL,
  full_name_zh VARCHAR(100),
  nationality VARCHAR(100),
  id_type VARCHAR(20) NOT NULL COMMENT 'passport/hkid/other',
  id_number VARCHAR(50) NOT NULL,
  date_of_birth DATE,
  
  -- 联系方式
  email VARCHAR(100),
  phone VARCHAR(20),
  residential_address TEXT,
  
  -- 持股/职务信息
  ownership_percentage DECIMAL(5,2) COMMENT '持股比例（实益拥有人）',
  position VARCHAR(100) COMMENT '职位（董事/授权人）',
  
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (corporate_account_id) REFERENCES corporate_accounts(id) ON DELETE CASCADE,
  INDEX idx_corporate_account (corporate_account_id),
  INDEX idx_party_type (party_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 子表3：corporate_documents (上传文件)

```sql
CREATE TABLE corporate_documents (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  corporate_account_id BIGINT NOT NULL,
  
  document_type VARCHAR(50) NOT NULL COMMENT 'incorporation_cert/business_reg/bank_proof/etc',
  file_name VARCHAR(255) NOT NULL,
  file_path VARCHAR(500) NOT NULL,
  file_size BIGINT,
  mime_type VARCHAR(100),
  
  uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  uploaded_by VARCHAR(100),
  
  FOREIGN KEY (corporate_account_id) REFERENCES corporate_accounts(id) ON DELETE CASCADE,
  INDEX idx_corporate_account (corporate_account_id),
  INDEX idx_document_type (document_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 子表4：corporate_risk_assessments (风险评估记录)

```sql
CREATE TABLE corporate_risk_assessments (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  corporate_account_id BIGINT NOT NULL,
  
  -- Part 1 通用问卷答案 (Q1-Q6)
  q1_holding_products JSON COMMENT '持有产品',
  q2_investment_horizon VARCHAR(20),
  q3_price_fluctuation VARCHAR(20),
  q4_net_worth_percentage VARCHAR(20),
  q5_attitude VARCHAR(20),
  q6_derivatives_knowledge JSON,
  
  -- Part 2B 机构客户问卷 (Q11-Q14)
  q11_investment_amount VARCHAR(20),
  q12_high_risk_percentage VARCHAR(20),
  q13_professional_staff VARCHAR(20),
  q14_liquid_assets VARCHAR(20),
  
  -- 评分结果
  total_score INT,
  risk_profile VARCHAR(20) NOT NULL COMMENT 'Conservative/Moderate/Aggressive',
  
  completed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (corporate_account_id) REFERENCES corporate_accounts(id) ON DELETE CASCADE,
  INDEX idx_corporate_account (corporate_account_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 子表5：corporate_declarations (监管声明与签署)

```sql
CREATE TABLE corporate_declarations (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  corporate_account_id BIGINT NOT NULL,
  
  agreement_type VARCHAR(50) NOT NULL COMMENT 'cash_account_agreement/risk_disclosure/etc',
  agreement_version VARCHAR(20),
  
  -- 电子签署信息
  signatory_name VARCHAR(100) NOT NULL COMMENT '授权签署人姓名',
  signatory_title VARCHAR(100) NOT NULL COMMENT '职衔',
  signature_data TEXT COMMENT '电子签名数据（Base64）',
  company_chop_data TEXT COMMENT '公司印章数据（Base64）',
  
  ip_address VARCHAR(50),
  user_agent TEXT,
  signed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  
  FOREIGN KEY (corporate_account_id) REFERENCES corporate_accounts(id) ON DELETE CASCADE,
  INDEX idx_corporate_account (corporate_account_id),
  INDEX idx_signed_at (signed_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```
