```typescript
function calculateRiskProfile(part1: Part1Answers, part2B: Part2BAnswers): RiskProfile {
  let score = 0;
  
  // Part 1 评分 (Q1-Q6)
  score += scoreQ1(part1.q1_holdingProducts);
  score += scoreQ2(part1.q2_investmentHorizon);
  score += scoreQ3(part1.q3_priceFluctuation);
  score += scoreQ4(part1.q4_netWorthPercentage);
  score += scoreQ5(part1.q5_attitude);
  score += scoreQ6(part1.q6_derivativesKnowledge);
  
  // Part 2B 评分 (Q11-Q14)
  score += scoreQ11(part2B.q11_investmentAmount);
  score += scoreQ12(part2B.q12_highRiskPercentage);
  score += scoreQ13(part2B.q13_professionalStaff);
  score += scoreQ14(part2B.q14_liquidAssets);
  
  if (score < 15) return 'Conservative';
  if (score < 30) return 'Moderate';
  return 'Aggressive';
}
```

```typescript
const validateDocuments = (docs: UploadedDocuments): ValidationResult => {
  const errors = [];
  
  if (!docs.incorporationCertificate) {
    errors.push('必须上传公司注册证书');
  }
  
  if (isHongKongCompany && !docs.businessRegistration) {
    errors.push('香港公司必须提供商业登记证');
  }
  
  if (docs.directorIdCopies.length < directors.length) {
    errors.push('所有董事必须上传身份证明');
  }
  
  return { valid: errors.length === 0, errors };
};
```

```typescript
const submitApplication = async () => {
  try {

      showErrors(validation.errors);
      return;

    const response = await api.post('/corporate-accounts', {
      ...applicationData,
      status: 'submitted',
      submittedAt: new Date(),
    });
    
    // 3. 上传文件
    await uploadDocuments(response.accountId, documents);
    
    // 4. 显示成功消息
    showSuccess({
      title: '申请已提交！',
      message: `您的申请编号是：${response.accountNumber}`,
      action: '查看申请进度',
      onAction: () => router.push(`/applications/${response.accountId}`),
    });
    
  } catch (error) {
    showError('提交失败，请重试');
  } finally {
    setSubmitting(false);
  }
};
```
